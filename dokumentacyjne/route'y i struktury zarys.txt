STRONA KATEGORII:
"/kategoria/$kategoria_id":{

	$KATEGORIA = ID przesłany przez zmienną 

	
	wyświetla wszystkie watki z danej kategorii
	SELECT watek.id, watek.nazwa, kategoria.nazwa, uzytkownik.nazwa, watek.data_zalozenia, watek.data_modyfikacji FROM watek
	INNER JOIN kategoria ON watek.kategoria_id = kategoria.id
	INNER JOIN uzytkownik ON watek.uzytkownik_id = uzytkownik.id
	WHERE watek.kategoria_id = $KATEGORIA
	ORDER BY watek.data_modyfikacji DESC

	foreach wyświetla wątki
		NAZWA (link do strony wątku /$kategoria.nazwa/$watek.nazwa   przesyła dalej $watek.id)
		NAZWA UZYTKOWNIKA
		DATA_ZALOZENIA
		DATA_MODYFIKACJI

	na dole button "stwórz wątek", który jest linkiem do /$kategoria.nazwa/dodajwatek przekazuje dalej $KATEGORIA
}

	$WATEK = ID przekazany przez zmienną
	$UZYTKOWNIK = id uzytkownika zalogowanego, jakkolwiek to będzie implementowane, może przez sesje?

	wyświetla wszystkie posty z danego wątku

	POSTY = "SELECT post.watek_id,uzytkownik.nazwa, post.tresc, post.data_wyslania FROM post

	WHERE watek_id = $WATEK"

	foreach wyświetla posty
	NAZWA UZYTKOWNIKA
	DATA WYSLANIA
	TRESC

		TRESC
	
	po wyslaniu:

		WHERE id = $WATEK;
